66  if(!thread_start) {
67  std::unique_lock<std::mutex> lock(thread_start_mutex);
68  thread_start_cv.wait(lock, [this]() { return thread_start; });
69  }
70  struct timespec last_time, cur_time;
71  clock_gettime(CLOCK_REALTIME, &last_time);
72 
73  while(!thread_shutdown) {
74  bool predicate_fired = false;
75  // Take the predicate lock before reading the predicate lists
76  std::unique_lock<std::mutex> predicates_lock(predicates.predicate_mutex);
77 
78  // one time predicates need to be evaluated only until they become true
79  for(auto& pred : predicates.one_time_predicates) {
80  if(pred != nullptr && (pred->first(*derived_this) == true)) {
81  predicate_fired = true;
82  // Copy the trigger pointer locally, so it can continue running without
83  // segfaulting even if this predicate gets deleted when we unlock predicates_lock
84  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(pred->second);
85  predicates_lock.unlock();
86  (*trigger)(*derived_this);
87  predicates_lock.lock();
88  // erase the predicate as it was just found to be true
89  pred.reset();
90  }
91  }
92 
93  // recurrent predicates are evaluated each time they are found to be true
94  for(auto& pred : predicates.recurrent_predicates) {
95  if(pred != nullptr && (pred->first(*derived_this) == true)) {
96  predicate_fired = true;
97  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(pred->second);
98  predicates_lock.unlock();
99  (*trigger)(*derived_this);
100  predicates_lock.lock();
101  }
102  }
103 
104  // transition predicates are only evaluated when they change from false to true
105  // We need to use iterators here because we need to iterate over two lists in parallel
106  auto pred_it = predicates.transition_predicates.begin();
107  auto pred_state_it = predicates.transition_predicate_states.begin();
108  while(pred_it != predicates.transition_predicates.end()) {
109  if(*pred_it != nullptr) {
110  //*pred_state_it is the previous state of the predicate at *pred_it
111  bool curr_pred_state = (*pred_it)->first(*derived_this);
112  if(curr_pred_state == true && *pred_state_it == false) {
113  predicate_fired = true;
114  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(
115  (*pred_it)->second);
116  predicates_lock.unlock();
117  (*trigger)(*derived_this);
118  predicates_lock.lock();
119  }
120  *pred_state_it = curr_pred_state;
121 
122  ++pred_it;
123  ++pred_state_it;
124  }
125  }
126 
127  if(predicate_fired) {
128  // update last time
129  clock_gettime(CLOCK_REALTIME, &last_time);
130  } else {
131  clock_gettime(CLOCK_REALTIME, &cur_time);
132  // check if the system has been inactive for enough time to induce sleep
133  double time_elapsed_in_ms = (cur_time.tv_sec - last_time.tv_sec) * 1e3
134  + (cur_time.tv_nsec - last_time.tv_nsec) / 1e6;
135  if(time_elapsed_in_ms > 1) {
136  predicates_lock.unlock();
137  using namespace std::chrono_literals;
138  std::this_thread::sleep_for(1ms);
139  predicates_lock.lock();
140  }
141  }
142  //Still to do: Clean up deleted predicates
143  }
