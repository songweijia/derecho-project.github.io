121 void test_provision_subgroups(const SubgroupInfo& subgroup_info,
122  const std::unique_ptr<View>& prev_view,

124  bool previous_was_ok = !prev_view || prev_view->is_adequately_provisioned;
125  int32_t initial_next_unassigned_rank = curr_view.next_unassigned_rank;
126  std::cout << "View has these members: " << curr_view.members << std::endl;
127  for(const auto& subgroup_type : subgroup_info.membership_function_order) {
128  subgroup_shard_layout_t subgroup_shard_views;
129  try {
130  auto temp = subgroup_info.subgroup_membership_functions.at(subgroup_type)(curr_view, curr_view.next_unassigned_rank, previous_was_ok);
131  subgroup_shard_views = std::move(temp);
132  std::cout << "Subgroup type " << subgroup_type.name() << " got assignment: " << std::endl;
133  derecho::print_subgroup_layout(subgroup_shard_views);
134  std::cout << "next_unassigned_rank is " << curr_view.next_unassigned_rank << std::endl
135  << std::endl;
136  } catch(derecho::subgroup_provisioning_exception& ex) {
137  curr_view.is_adequately_provisioned = false;
138  curr_view.next_unassigned_rank = initial_next_unassigned_rank;
139  curr_view.subgroup_shard_views.clear();
140  std::cout << "Subgroup type " << subgroup_type.name() << " failed to provision, marking View inadequate" << std::endl
141  << std::endl;

147  //Assign this (type, index) pair a new unique subgroup ID

179 std::unique_ptr<View> make_next_view(const View& curr_view,
180  const std::set<int>& leave_ranks,
181  const std::vector<node_id_t>& joiner_ids,
182  const std::vector<ip_addr>& joiner_ips) {
183  int next_num_members = curr_view.num_members - leave_ranks.size() + joiner_ids.size();
184  std::vector<node_id_t> joined, members(next_num_members), departed;
185  std::vector<char> failed(next_num_members);
186  std::vector<ip_addr> member_ips(next_num_members);
187  int next_unassigned_rank = curr_view.next_unassigned_rank;
188  for(std::size_t i = 0; i < joiner_ids.size(); ++i) {
189  joined.emplace_back(joiner_ids[i]);
190  //New members go at the end of the members list, but it may shrink in the new view
191  int new_member_rank = curr_view.num_members - leave_ranks.size() + i;
192  members[new_member_rank] = joiner_ids[i];
193  member_ips[new_member_rank] = joiner_ips[i];
194  }
195  for(const auto& leaver_rank : leave_ranks) {
196  departed.emplace_back(curr_view.members[leaver_rank]);
197  //Decrement next_unassigned_rank for every failure, unless the failure wasn't in a subgroup anyway
198  if(leaver_rank <= curr_view.next_unassigned_rank) {
199  next_unassigned_rank--;
200  }
201  }
202  //Copy member information, excluding the members that have failed
203  int m = 0;
204  for(int n = 0; n < curr_view.num_members; n++) {
205  //This is why leave_ranks needs to be a set
206  if(leave_ranks.find(n) == leave_ranks.end()) {
207  members[m] = curr_view.members[n];
208  member_ips[m] = curr_view.member_ips[n];
209  failed[m] = curr_view.failed[n];
210  ++m;
211  }
212  }
213 
214  //Initialize my_rank in next_view
215  int32_t my_new_rank = -1;
216  node_id_t myID = curr_view.members[curr_view.my_rank];
217  for(int i = 0; i < next_num_members; ++i) {
218  if(members[i] == myID) {
219  my_new_rank = i;
220  break;
221  }
222  }
223  return std::make_unique<View>(curr_view.vid + 1, members, member_ips, failed,
224  joined, departed, my_new_rank, next_unassigned_rank);
225 }
