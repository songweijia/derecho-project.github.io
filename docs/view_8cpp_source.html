184 void View::merge_changes() {
185  int myRank = my_rank;
186  // Merge the change lists
187  for(int n = 0; n < num_members; n++) {
188  if(gmsSST->num_changes[myRank] < gmsSST->num_changes[n]) {
189  gmssst::set(gmsSST->changes[myRank], gmsSST->changes[n], gmsSST->changes.size());
190  gmssst::set(gmsSST->num_changes[myRank], gmsSST->num_changes[n]);
191  }
192 
193  if(gmsSST->num_committed[myRank] < gmsSST->num_committed[n]) // How many I know to have been committed
194  {
195  gmssst::set(gmsSST->num_committed[myRank], gmsSST->num_committed[n]);
196  }
197  }
198  bool found = false;
199  for(int n = 0; n < num_members; n++) {
200  if(failed[n]) {
201  // Make sure that the failed process is listed in the changes vector as a proposed change
202  for(int c = gmsSST->num_committed[myRank]; c < gmsSST->num_changes[myRank] && !found; c++) {
203  if(gmsSST->changes[myRank][c % gmsSST->changes.size()] == members[n]) {
204  // Already listed
205  found = true;
206  }
207  }
208  } else {
209  // Not failed
210  found = true;
211  }
212 
213  if(!found) {
214  gmssst::set(gmsSST->changes[myRank][gmsSST->num_changes[myRank] % gmsSST->changes.size()],
215  members[n]);
216  gmssst::increment(gmsSST->num_changes[myRank]);
217  }
218  }
219  gmsSST->put(gmsSST->changes.get_base() - gmsSST->getBaseAddress(),
220  gmsSST->num_acked.get_base() - gmsSST->changes.get_base());
221 }

258  std::ifstream view_file_swap(view_file_name + ".swp");
259  std::unique_ptr<View> view;
260  std::unique_ptr<View> swap_view;
261  //The expected view file might not exist, in which case we'll fall back to the swap file
262  if(view_file.good()) {
263  //Each file contains the size of the view (an int copied as bytes),
264  //followed by a serialized view
265  std::size_t size_of_view;
266  view_file.read((char*)&size_of_view, sizeof(size_of_view));
267  char buffer[size_of_view];
268  view_file.read(buffer, size_of_view);
269  //If the view file doesn't contain a complete view (due to a crash
270  //during writing), the read() call will set failbit
271  if(!view_file.fail()) {
272  view = mutils::from_bytes<View>(nullptr, buffer);
273  }
274  }
275  if(view_file_swap.good()) {
276  std::size_t size_of_view;
277  view_file_swap.read((char*)&size_of_view, sizeof(size_of_view));
278  char buffer[size_of_view];
279  view_file_swap.read(buffer, size_of_view);
280  if(!view_file_swap.fail()) {
281  swap_view = mutils::from_bytes<View>(nullptr, buffer);
282  }
283  }
284  if(swap_view == nullptr || (view != nullptr && view->vid >= swap_view->vid)) {
285  return view;
286  } else {

305 View parse_view(std::istream& stream) {
306  std::string line;
307  int32_t vid = 0;
308  if(std::getline(stream, line)) {
309  vid = std::stoi(line);
310  }
311  std::vector<node_id_t> members;
312  //"List of member IDs" line
313  if(std::getline(stream, line)) {
314  std::istringstream linestream(line);
315  std::copy(std::istream_iterator<node_id_t>(linestream), std::istream_iterator<node_id_t>(),
316  std::back_inserter(members));
317  }
318  std::vector<ip_addr> member_ips;
319  //"List of member IPs" line
320  if(std::getline(stream, line)) {
321  std::istringstream linestream(line);
322  std::copy(std::istream_iterator<ip_addr>(linestream), std::istream_iterator<ip_addr>(),
323  std::back_inserter(member_ips));
324  }
325  std::vector<char> failed;
326  //Failures array line, which was printed as "T" or "F" strings
327  if(std::getline(stream, line)) {
328  std::istringstream linestream(line);
329  std::string fail_str;
330  while(linestream >> fail_str) {
331  failed.emplace_back(fail_str == "T" ? true : false);
332  }
333  }
334  int32_t my_rank = -1;
335  if(std::getline(stream, line)) {
336  my_rank = std::stoi(line);
337  }
338  return View(vid, members, member_ips, failed, {}, {}, my_rank);
339 }
