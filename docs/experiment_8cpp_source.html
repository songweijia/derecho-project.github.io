64  atomic<uint64_t> end_time;

72  uint16_t base_group_number = next_group_number;

81  base_group_number + i, members, block_size, type,
82  [&mr, i, buffer_size](size_t length) -> rdmc::receive_destination {
83  return {mr, buffer_size * i};
84  },
85  [&](char *data, size_t) {
86  if(--sends_remaining == 0) {

88  end_ptime = get_process_time();

91  send_done_cv.notify_all();
92  }

107  end_ptime = 0;

123

576  size_t n = 0;
577  auto t = get_time();
578  for(size_t group_size = 2; group_size <= 64; group_size++) {
579  for(size_t message_size = 1; message_size <= 32; message_size++) {
580  size_t total_steps = message_size + ceil(log2(group_size)) - 1;
581  for(unsigned int step = 0; step < total_steps; step++) {
582  for(unsigned int node = 0; node < group_size; node++) {
583  // Compute the outgoing transfer for this node/step
584  auto transfer = binomial_schedule::get_outgoing_transfer(
585  node, step, group_size, floor(log2(group_size)),
586  message_size, total_steps);
587  n++;
588 
589  if(transfer) {
590  // See what the supposed sender is doing this step
591  auto reverse = binomial_schedule::get_incoming_transfer(
592  transfer->target, step, group_size,
593  floor(log2(group_size)), message_size, total_steps);
594  n++;
595 
596  // Make sure the two nodes agree
597  if(!reverse) throw false;
598  if(transfer->block_number != reverse->block_number)
599  throw false;
600 
601  // If we aren't the root sender, also check that the
602  // node got this block on a past step.
603  if(node != 0) {
604  for(int s = step - 1; s >= 0; s--) {
605  auto prev = binomial_schedule::get_incoming_transfer(
606  node, s, group_size,
607  floor(log2(group_size)), message_size,
608  total_steps);
609  n++;
610  if(prev && prev->block_number == transfer->block_number)
611  break;
612 
613  if(s == 0) {
614  throw false;
615  }
616  }
617  }
618  }
619 
620  // Compute the incoming transfer for this node/step
621  transfer = binomial_schedule::get_incoming_transfer(
622  node, step, group_size, floor(log2(group_size)),
623  message_size, total_steps);
624  n++;
625 
626  if(transfer) {
627  // Again make sure the supposed receiver agrees
628  auto reverse = binomial_schedule::get_outgoing_transfer(
629  transfer->target, step, group_size,
630  floor(log2(group_size)), message_size, total_steps);
631  n++;
632  if(!reverse) throw false;
633  if(transfer->block_number != reverse->block_number)
634  throw false;
635  if(reverse->target != node) throw false;
636 
637  // Make sure we don't already have the block we're
638  // getting.
639  for(int s = step - 1; s >= 0; s--) {
640  auto prev = binomial_schedule::get_incoming_transfer(
641  node, s, group_size, floor(log2(group_size)),
642  message_size, total_steps);
643  n++;
644  if(prev && prev->block_number == transfer->block_number) {
645  throw false;
646  }
647  }
648  }
649  }
650  }
651 
652  // Make sure that all nodes get every block
653  for(unsigned int node = 1; node < group_size; node++) {
654  set<size_t> blocks;
655  for(unsigned int step = 0; step < total_steps; step++) {
656  auto transfer = binomial_schedule::get_incoming_transfer(
657  node, step, group_size, floor(log2(group_size)),
658  message_size, total_steps);
659  n++;
660 
661  if(transfer) blocks.insert(transfer->block_number);
662  }
663  if(blocks.size() != message_size) throw false;
664  }
665  }
666  }
667  auto diff = get_time() - t;
668  printf("average time = %f ns\n", (double)diff / n);
669  puts("PASS");
670 }
