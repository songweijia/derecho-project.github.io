556  bool msgs_delivered = false;
557  // DERECHO_LOG(-1, -1, "deliver_messages_upto");

561  int32_t max_seq_num = curr_seq_num;

563  max_seq_num = std::max(max_seq_num,
564  static_cast<int32_t>(max_indices_for_senders[sender] * num_shard_senders + sender));
565  }
566  // DERECHO_LOG(-1, -1, "deliver_messages_upto_loop");
567  for(int32_t seq_num = curr_seq_num + 1; seq_num <= max_seq_num; seq_num++) {
568  //determine if this sequence number should actually be skipped
569  int32_t index = seq_num / num_shard_senders;
570  uint32_t sender_rank = seq_num % num_shard_senders;
571  if(index > max_indices_for_senders[sender_rank]) {
572  continue;
573  }

575  if(rdmc_msg_ptr != locally_stable_rdmc_messages[subgroup_num].end()) {
576  auto& msg = rdmc_msg_ptr->second;

578  uint64_t msg_ts = ((header*) buf)->timestamp;
579  msgs_delivered = true;
580  //Note: deliver_message frees the RDMC buffer in msg, which is why the timestamp must be saved before calling this
581  deliver_message(msg, subgroup_num);

590  uint64_t msg_ts = ((header*) buf)->timestamp;

592  version_message(msg, subgroup_num, seq_num, msg_ts);
593  locally_stable_sst_messages[subgroup_num].erase(seq_num);
594  }

597  sst->put(get_shard_sst_indices(subgroup_num),
598  (char*)std::addressof(sst->delivered_num[0][subgroup_num]) - sst->getBaseAddress(),
599  sizeof(decltype(sst->delivered_num)::value_type));
600  if(subgroup_settings.at(subgroup_num).mode != Mode::UNORDERED && msgs_delivered) {
601  //Call the persistence_manager_post_persist_func
602  std::get<1>(persistence_manager_callbacks)(subgroup_num,
603  persistent::combine_int32s(sst->vid[member_index], sst->delivered_num[member_index][subgroup_num]));

662 void MulticastGroup::sst_receive_handler(subgroup_id_t subgroup_num, const SubgroupSettings& curr_subgroup_settings,
663  const std::map<uint32_t, uint32_t>& shard_ranks_by_sender_rank,
664  uint32_t num_shard_senders, uint32_t sender_rank,
665  volatile char* data, uint32_t size) {
666  header* h = (header*)data;
667  int32_t index = h->index;
668  // std::cout << "subgroup_num=" << subgroup_num << ", sender_rank=" << sender_rank << ", index=" << index << std::endl;
669  auto beg_index = index;
670  message_id_t sequence_number = index * num_shard_senders + sender_rank;
671  whenlog(logger->trace("Locally received message in subgroup {}, sender rank {}, index {}. Header fields: header_size={}, pause_sending_turns={}, index={}, timestamp={}", subgroup_num, sender_rank, index, h->header_size, h->pause_sending_turns, h->index, h->timestamp);)
672 
673  node_id_t node_id = curr_subgroup_settings.members[shard_ranks_by_sender_rank.at(sender_rank)];
674 
675  // DERECHO_LOG(node_id, index, "received_message");
676  locally_stable_sst_messages[subgroup_num][sequence_number] = {node_id, index, size, data};
677 
678  // Add empty messages to locally_stable_sst_messages for each turn that the sender is skipping.
679  for(unsigned int j = 0; j < h->pause_sending_turns; ++j) {
680  index++;
681  sequence_number += num_shard_senders;
682  locally_stable_sst_messages[subgroup_num][sequence_number] = {node_id, index, 0, 0};
683  }
684 
685  auto new_num_received = resolve_num_received(beg_index, index, curr_subgroup_settings.num_received_offset + sender_rank);
686  if(curr_subgroup_settings.mode == Mode::UNORDERED) {
687  // issue stability upcalls for the recently sequenced messages
688  for(int i = sst->num_received[member_index][curr_subgroup_settings.num_received_offset + sender_rank] + 1; i <= new_num_received; ++i) {
689  message_id_t seq_num = i * num_shard_senders + sender_rank;
690  if(!locally_stable_sst_messages[subgroup_num].empty()
691  && locally_stable_sst_messages[subgroup_num].begin()->first == seq_num) {
692  auto& msg = locally_stable_sst_messages[subgroup_num].begin()->second;
693  if(msg.size > 0) {
694  char* buf = const_cast<char*>(msg.buf);
695  header* h = (header*)(buf);
696  if(callbacks.global_stability_callback) {
697  callbacks.global_stability_callback(subgroup_num, msg.sender_id,
698  msg.index, buf + h->header_size,
699  msg.size - h->header_size);
700  }
701  if(node_id == members[member_index]) {
702  pending_message_timestamps[subgroup_num].erase(h->timestamp);
703  }
704  }
705  locally_stable_sst_messages[subgroup_num].erase(locally_stable_sst_messages[subgroup_num].begin());
706  } else {
707  assert(!locally_stable_rdmc_messages[subgroup_num].empty());
708  auto it2 = locally_stable_rdmc_messages[subgroup_num].begin();
709  assert(it2->first == seq_num);
710  auto& msg = it2->second;
711  if(msg.size > 0) {
712  char* buf = msg.message_buffer.buffer.get();
713  header* h = (header*)(buf);
714  if(callbacks.global_stability_callback) {
715  callbacks.global_stability_callback(subgroup_num, msg.sender_id,
716  msg.index, buf + h->header_size,
717  msg.size - h->header_size);
718  }
719  free_message_buffers[subgroup_num].push_back(std::move(msg.message_buffer));
720  if(node_id == members[member_index]) {
721  pending_message_timestamps[subgroup_num].erase(h->timestamp);
722  }
723  }
724  locally_stable_rdmc_messages[subgroup_num].erase(it2);
725  }
726  }
727  }
728  sst->num_received[member_index][curr_subgroup_settings.num_received_offset + sender_rank] = new_num_received;
729 }

731 void MulticastGroup::receiver_function(subgroup_id_t subgroup_num, const SubgroupSettings& curr_subgroup_settings,
732  const std::map<uint32_t, uint32_t>& shard_ranks_by_sender_rank,
733  uint32_t num_shard_senders, DerechoSST& sst, unsigned int batch_size,
734  const std::function<void(uint32_t, volatile char*, uint32_t)>& sst_receive_handler_lambda) {
735  // DERECHO_LOG(receiver_cnt, -1, "in receiver_trig");
736  std::lock_guard<std::mutex> lock(msg_state_mtx);
737  for(uint i = 0; i < batch_size; ++i) {
738  for(uint sender_count = 0; sender_count < num_shard_senders; ++sender_count) {
739  auto num_received = sst.num_received_sst[member_index][curr_subgroup_settings.num_received_offset + sender_count] + 1;

742  if(next_seq == num_received / static_cast<int32_t>(window_size) + 1) {
743  whenlog(logger->trace("receiver_trig calling sst_receive_handler_lambda. next_seq = {}, num_received = {}, sender rank = {}. Reading from SST row {}, slot {}",
744  next_seq, num_received, sender_count, node_id_to_sst_index.at(curr_subgroup_settings.members[shard_ranks_by_sender_rank.at(sender_count)]), (subgroup_num * window_size + slot));)
745  sst_receive_handler_lambda(sender_count,
746  sst.slots[node_id_to_sst_index.at(curr_subgroup_settings.members[shard_ranks_by_sender_rank.at(sender_count)])][subgroup_num * window_size + slot].buf,
747  sst.slots[node_id_to_sst_index.at(curr_subgroup_settings.members[shard_ranks_by_sender_rank.at(sender_count)])][subgroup_num * window_size + slot].size);
748  sst.num_received_sst[member_index][curr_subgroup_settings.num_received_offset + sender_count] = num_received;
749  }
750  }
751  }
752  sst.put((char*)std::addressof(sst.num_received_sst[0][curr_subgroup_settings.num_received_offset]) - sst.getBaseAddress(),
753  sizeof(decltype(sst.num_received_sst)::value_type) * num_shard_senders);
754  // std::atomic_signal_fence(std::memory_order_acq_rel);
755  auto* min_ptr = std::min_element(&sst.num_received[member_index][curr_subgroup_settings.num_received_offset],
756  &sst.num_received[member_index][curr_subgroup_settings.num_received_offset + num_shard_senders]);

759  if(new_seq_num > sst.seq_num[member_index][subgroup_num]) {
760  whenlog(logger->trace("Updating seq_num for subgroup {} to {}", subgroup_num, new_seq_num);)
761  sst.seq_num[member_index][subgroup_num] = new_seq_num;
762  sst.put((char*)std::addressof(sst.seq_num[0][subgroup_num]) - sst.getBaseAddress(),
763  sizeof(decltype(sst.seq_num)::value_type));
764  }
765  sst.put((char*)std::addressof(sst.num_received[0][curr_subgroup_settings.num_received_offset]) - sst.getBaseAddress(),
766  sizeof(decltype(sst.num_received)::value_type) * num_shard_senders);
767 }

769 void MulticastGroup::delivery_trigger(subgroup_id_t subgroup_num, const SubgroupSettings& curr_subgroup_settings,
770  const uint32_t num_shard_members, DerechoSST& sst) {
771  // DERECHO_LOG(delivery_cnt, -1, "in delivery_trig");
772  std::lock_guard<std::mutex> lock(msg_state_mtx);
773  // compute the min of the stable_num
774  message_id_t min_stable_num
775  = sst.stable_num[node_id_to_sst_index.at(curr_subgroup_settings.members[0])][subgroup_num];
776  for(uint i = 0; i < num_shard_members; ++i) {
777  if(sst.stable_num[node_id_to_sst_index.at(curr_subgroup_settings.members[i])][subgroup_num] < min_stable_num) {
778  min_stable_num = sst.stable_num[node_id_to_sst_index.at(curr_subgroup_settings.members[i])][subgroup_num];
779  }
780  }
781 
782  bool update_sst = false;
783  while(true) {
784  if(locally_stable_rdmc_messages[subgroup_num].empty() && locally_stable_sst_messages[subgroup_num].empty()) {
785  break;
786  }
787  int32_t least_undelivered_rdmc_seq_num, least_undelivered_sst_seq_num;
788  least_undelivered_rdmc_seq_num = least_undelivered_sst_seq_num = std::numeric_limits<int32_t>::max();

793  least_undelivered_sst_seq_num = locally_stable_sst_messages[subgroup_num].begin()->first;

800  if(msg.size > 0) {

804  deliver_message(msg, subgroup_num);
805  version_message(msg, subgroup_num, least_undelivered_rdmc_seq_num, msg_ts);
806  }
807  // DERECHO_LOG(-1, -1, "deliver_message() done");
808  sst.delivered_num[member_index][subgroup_num] = least_undelivered_rdmc_seq_num;
809  locally_stable_rdmc_messages[subgroup_num].erase(locally_stable_rdmc_messages[subgroup_num].begin());
810  // DERECHO_LOG(-1, -1, "message_erase_done");
811  } else if(least_undelivered_sst_seq_num < least_undelivered_rdmc_seq_num && least_undelivered_sst_seq_num <= min_stable_num) {
812  update_sst = true;
813  whenlog(logger->trace("Subgroup {}, can deliver a locally stable SST message: min_stable_num={} and least_undelivered_seq_num={}",
814  subgroup_num, min_stable_num, least_undelivered_sst_seq_num);)
815  SSTMessage& msg = locally_stable_sst_messages[subgroup_num].begin()->second;
816  if(msg.size > 0) {
817  char* buf = (char*)msg.buf;
818  uint64_t msg_ts = ((header*) buf)->timestamp;
819  deliver_message(msg, subgroup_num);
820  version_message(msg, subgroup_num, least_undelivered_sst_seq_num, msg_ts);
821  }
822  // DERECHO_LOG(-1, -1, "deliver_message() done");
823  sst.delivered_num[member_index][subgroup_num] = least_undelivered_sst_seq_num;
824  locally_stable_sst_messages[subgroup_num].erase(locally_stable_sst_messages[subgroup_num].begin());
825  // DERECHO_LOG(-1, -1, "message_erase_done");
826  } else {
827  break;
828  }
829  }
830  if(update_sst) {
831  // DERECHO_LOG(-1, -1, "delivery_put_start");
832  sst.put(get_shard_sst_indices(subgroup_num),
833  (char*)std::addressof(sst.delivered_num[0][subgroup_num]) - sst.getBaseAddress(),
834  sizeof(decltype(sst.delivered_num)::value_type));
835  // locally_stable_messages[subgroup_num].erase(locally_stable_messages[subgroup_num].begin());
836  //post persistence request for ordered mode.
837  if(curr_subgroup_settings.mode != Mode::UNORDERED) {
838  std::get<1>(persistence_manager_callbacks)(subgroup_num,
839  persistent::combine_int32s(sst.vid[member_index], sst.delivered_num[member_index][subgroup_num]));
840  }
