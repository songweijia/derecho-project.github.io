106  if(polling_loop_shutdown_flag) return;

108  }

112  void *ev_ctx;
113  ibv_get_cq_event(verbs_resources.cc, &ev_cq, &ev_ctx);
114  ibv_ack_cq_events(ev_cq, 1);
115  }
116  }
117  }
118  }

121  fprintf(stderr, "Failed to poll completion queue.");
122  continue;
123  }

130  if(wc.status != 0) {
131  string opcode = "[unknown]";
132  if(wc.opcode == IBV_WC_SEND) opcode = "IBV_WC_SEND";
133  if(wc.opcode == IBV_WC_RECV) opcode = "IBV_WC_RECV";
134  if(wc.opcode == IBV_WC_RDMA_WRITE) opcode = "IBV_WC_RDMA_WRITE";

403 bool verbs_add_connection(uint32_t index, const string &address,
404  uint32_t node_rank) {

406  if(sockets.count(index) > 0) {
407  fprintf(stderr,
408  "WARNING: attempted to connect to node %u at %s:%d but we "
409  "already have a connection to a node with that index.",
410  (unsigned int)index, address.c_str(), derecho::rdmc_tcp_port);
411  return false;
412  }
413 
414  try {
415  sockets[index] = tcp::socket(address, derecho::rdmc_tcp_port);
416  } catch(tcp::exception) {
417  fprintf(stderr, "WARNING: failed to node %u at %s:%d",
418  (unsigned int)index, address.c_str(), derecho::rdmc_tcp_port);
419  return false;
420  }
421 
422  // Make sure that the connection works, and that we've connected to the
423  // right node.
424  uint32_t remote_rank = 0;
425  if(!sockets[index].exchange(node_rank, remote_rank)) {
426  fprintf(stderr,
427  "WARNING: failed to exchange rank with node %u at %s:%d",
428  (unsigned int)index, address.c_str(), derecho::rdmc_tcp_port);
429  sockets.erase(index);
430  return false;
431  } else if(remote_rank != index) {
432  fprintf(stderr,
433  "WARNING: node at %s:%d replied with wrong rank (expected"
434  "%d but got %d)",
435  address.c_str(), derecho::rdmc_tcp_port, (unsigned int)index,
436  (unsigned int)remote_rank);
437 
438  sockets.erase(index);
439  return false;
440  }
441  return true;
442  } else if(index > node_rank) {
443  try {
444  tcp::socket s = connection_listener->accept();
445 
446  uint32_t remote_rank = 0;
447  if(!s.exchange(node_rank, remote_rank)) {
448  fprintf(stderr, "WARNING: failed to exchange rank with node");
449  return false;
450  } else {
451  sockets[remote_rank] = std::move(s);
452  return true;
453  }
454  } catch(tcp::exception) {
455  fprintf(stderr, "Got error while attempting to listing on port");
456  return false;
457  }
458  }
459 
460  return false; // we can't connect to ourselves...
461 }

979 bool task::post() {
980  size_t num_tasks = 1 + impl->send_list.size() + impl->recv_list.size();
981  auto tasks = make_unique<ibv_exp_task[]>(num_tasks);
982 
983  size_t index = 0;
984  for(auto &&l : impl->recv_list) {
985  for(size_t i = 0; i + 1 < l.second.size(); i++) {
986  impl->recv_wrs[l.second[i]].next = &impl->recv_wrs[l.second[i + 1]];
987  }
988 
989  tasks[index].item.qp = l.first;
990  tasks[index].item.recv_wr = &impl->recv_wrs[l.second.front()];
991  tasks[index].task_type = IBV_EXP_TASK_RECV;
992  tasks[index].next = &tasks[index + 1];
993  tasks[index].comp_mask = 0;
994  ++index;
995  }
996  for(auto &&l : impl->send_list) {
997  for(size_t i = 0; i + 1 < l.second.size(); i++) {
998  impl->send_wrs[l.second[i]].next = &impl->send_wrs[l.second[i + 1]];
999  }
1000 
1001  tasks[index].item.qp = l.first;
1002  tasks[index].item.send_wr = &impl->send_wrs[l.second.front()];
1003  tasks[index].task_type = IBV_EXP_TASK_SEND;
1004  tasks[index].next = &tasks[index + 1];
1005  tasks[index].comp_mask = 0;
1006  ++index;
1007  }
1008 
1009  for(size_t i = 0; i + 1 < impl->mqp_list.size(); i++) {
1010  impl->send_wrs[impl->mqp_list[i]].next = &impl->send_wrs[impl->mqp_list[i + 1]];
1011  }
1012 
1013  tasks[index].item.qp = impl->mqp;
1014  tasks[index].item.send_wr = &impl->send_wrs[impl->mqp_list.front()];
1015  tasks[index].task_type = IBV_EXP_TASK_SEND;
1016  tasks[index].next = nullptr;
1017  tasks[index].comp_mask = 0;
1018 
1019  ibv_exp_task *bad = nullptr;
1020  return !ibv_exp_post_task(verbs_resources.ib_ctx, &tasks[0], &bad);
1021 }
